!file_get(file_name): {
   path = process.current_exe()
   stream = iostream.file.open(file_name)
   stream.failure() -> return N
   stream.success() -> {
       return stream.collect().to_array()
   }
}

!random_password(): {
    temp = {}
    %num: {
        rand_int = random.int(0, string.get_combinations().size)
        rand_char = string.get_combinations()[rand_int]
        temp.append(rand_char)
    }

    password_string = string.string_builder.build(temp, "")
    return password_string
}

!main(): {
    num = _
    %: {
        print("Enter how many numbers of password to generate")
        num = iostream.get_int()
        num.failure() -> {
            print("Not a valid integer")
            continue
        }
    }

    print("Your new password is ", F) -> print(random_password())
    
    process.pause()
}

!is_open_rblx(): {

        ? return F
    }
